### Forma 1 - ngIF
```html
<div class="container">
  <div class="form-box">
    <form #frm="ngForm" (ngSubmit)="calculaResultado()">
      <h2>Calculadora</h2>
      <label for="numero1">Número 1</label>
      <input type="number" id="numero1" name="numero1" [(ngModel)]="numero1" />
      <label for="numero2">Número 2</label>
      <input type="number" id="numero2" name="numero2" [(ngModel)]="numero2" />
      <button type="submit">Somar</button>
    </form>

    @if (resultado) {
      <div class="resultado-display">
        <p>
          Resultado: <span id="result">{{ resultado }}</span>
        </p>
      </div>
    }
  </div>
</div>
```

### Forma 2 - CommomModules

```typescript
import { CommonModule } from '@angular/common';
import { Component } from '@angular/core';
import { FormsModule } from '@angular/forms';

@Component({
  selector: 'app-calculadora',
  imports: [FormsModule, CommonModule],
  templateUrl: './calculadora.component.html',
  styleUrl: './calculadora.component.scss',
})

export class CalculadoraComponent {
  numero1: number = 0; // Poderiamos inserir uma interrogacao logo após o nome da var para tornar a declaracao do valor opcional.
  numero2: number = 0;
  resultado: number = 0;
  
  calculaResultado() {
    console.log('Chamando o método para calcular o resultado!');
    console.log('Numero1: ' + this.numero1);
    console.log('Numero2: ' + this.numero2);
    this.resultado = this.numero1 + this.numero2;
  }
}
```
